The `useValue` subscribes to and returns the value of an atom or selector.

```jsx
import { atom, useValue, useAtom } from "valdres-react"

const time = atom(0)
const running = atom(false)

const Comp = () => {
    const [time, setTime] = useValue(time)
    const [isRunning, setIsRunning] = useAtom(running)

    useEffect(() => {
        setInterval(() => {})
    }, [isRunning])

    return (
        <div>
            <button onClick={() => setIsRunning(curr => !curr)}>
                {isRunning ? "Start" : "Stop"}
            </button>
            <button
                onClick={() => {
                    setIsRunning(false)
                    setTime(0)
                }}
            >
                Reset
            </button>
            {count}
        </div>
    )
}
```
