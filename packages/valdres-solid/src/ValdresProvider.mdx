Wraps a Solid component tree with a valdres store context.

## Usage

```tsx
import { ValdresProvider } from "valdres-solid"

function App() {
    return (
        <ValdresProvider>
            <Counter />
        </ValdresProvider>
    )
}
```

## With a custom store

```tsx
import { store } from "valdres"
import { ValdresProvider } from "valdres-solid"

const myStore = store()

function App() {
    return (
        <ValdresProvider store={myStore}>
            <Counter />
        </ValdresProvider>
    )
}
```
